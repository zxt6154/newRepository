Consumer()消费者
          void accept(T t) 意为消费一个指定泛型的数据
            default Consumer<T> andThen(Consumer<? super T> after
                 用法举例：                 

                       public static <T> void doJob(T t, Consumer<T> con){

                                                 con.accept(t);

                                          }

                       public static <T> void doJob(T t, Consumer<T> con1, Consumer<T> con2) {

                                con1.andThen(con2).accept(t);

                                      }
