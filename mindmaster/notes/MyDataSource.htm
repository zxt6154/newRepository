public class MyDataSource implements DataSource {
    //创建一个保存连接的容器
    /* arraylist 查询快，增删慢
        linkedList 查询慢，增删快*/
private LinkedList<Connection> pools=new LinkedList<>();
    public MyDataSource() {
        for (int i=0;i<5;i++){
            Connection conn=JDBCUtils.getConnection();
            pools.addFirst(conn);
        }
    }

    @Override
    public Connection getConnection() throws SQLException {
        if (pools.size()==0){
            for (int i = 0; i < 3; i++) {
                Connection conn=JDBCUtils.getConnection();
                pools.addFirst(conn);
            }
        }
        Connection conn = pools.removeLast();

        // 使用动态代理对 Connection 增强close方法，实现将连接放回连接池
        ClassLoader loader = conn.getClass().getClassLoader();
        Class<?>[] interfaces=new Class[]{Connection.class};
        //Class<?>[] jikou = conn.getClass().getInterfaces();
        System.out.println(interfaces);
        InvocationHandler h = new InvocationHandler(){
            @Override
            public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
                if ("close".equals(method.getName())){
                    backToPool(conn);
                    return null;
                }
                return method.invoke(conn,args);
            }
        };
        

        Connection connProxy= (Connection) Proxy.newProxyInstance(loader,interfaces,h);
        return conn;
    }

    private void backToPool(Connection conn) {
        pools.addFirst(conn);
        System.out.println("将对应连接放回连接池");
    }

    @Override
    public Connection getConnection(String username, String password) throws SQLException {
        return null;
    }

    @Override
    public <T> T unwrap(Class<T> iface) throws SQLException {
        return null;
    }

    @Override
    public boolean isWrapperFor(Class<?> iface) throws SQLException {
        return false;
    }

    @Override
    public PrintWriter getLogWriter() throws SQLException {
        return null;
    }

    @Override
    public void setLogWriter(PrintWriter out) throws SQLException {

    }

    @Override
    public void setLoginTimeout(int seconds) throws SQLException {

    }

    @Override
    public int getLoginTimeout() throws SQLException {
        return 0;
    }

    @Override
    public Logger getParentLogger() throws SQLFeatureNotSupportedException {
        return null;
    }
}
