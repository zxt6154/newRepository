public class PredicateTest{
    public static void main(String[] args) {
        /*
         * a)是否是自然数(大于等于0);
         * b)绝对值是否大于100
         * c)是否是偶数
         * 遍历arr，仅利用已创建的Predicate对象(不使用任何逻辑运算符)，完成以下需求

         * */
        Integer[] arr = {-12345, 9999, 520, 0, -38, -7758520, 941213};
        System.out.println("----------------i.打印自然数----------------------");
        System.out.println("stream方法");
        Stream<Integer> stream = Stream.of(arr);
        Stream<Integer> stre = stream.filter(num -> num >= 0);
        stre.forEach(System.out::println);
        System.out.println();
// ii.打印负整数的个数(<0)
        System.out.println("----------------i.打印负整数----------------------");
        test01(n -> n < 0, arr);
        System.out.println();
        System.out.println("--------------------------------");
        test02(i->i>100,
                i->i%2==0,
                arr
                );
// iv.打印是负整数或偶数的数的个数
        System.out.println("----------打印是负整数或偶数的数的个数----------");
        test03(n -> n % 2 == 0,
                n -> n < 0,
                arr);
    }

    // ii.打印负整数的个数(<0)
    public static void test01(Predicate<Integer> pre, Integer[] num) {
        for (Integer i : num) {
            if (pre.test(i))
                System.out.print(i + " ");
        }
    }

    // iii.打印绝对值大于100的偶数的个数 绝对值大于100 + 偶数
   public static void test02(Predicate<Integer> pre1,Predicate<Integer> pre2,Integer[] num){
       for (Integer i : num) {
                   i=i>0?i:-i;
                   if (pre1.and(pre2).test(i))
                       System.out.println(i);
           //Predicate<Integer> and = pre1.and(pre2);
       }
   }

    // iv.打印是负整数或偶数的数的个数
    public static void test03(Predicate<Integer> pre1, Predicate<Integer> pre2, Integer[] num) {
        for (Integer i : num) {
            if (pre1.or(pre2).test(i))
                System.out.print(i + " ");
        }
//
    }
}
