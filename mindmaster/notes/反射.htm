           获取类的class对象:1. Class c=类.class；2.Class c=Class.forName("类的全限定名")；3.Class c=对象名.getClass();
           根据获取的class对象获取其构造方法:
                                            Constructor getConstructor(需要获取的public构造方法的参数列表.Class);
                                            Constructor getDeclaredConstructor(需要获取的private构造方法的参数列表.Class);
                                            void setAccessible(true/false) 暴力反射(权限检查)，设置可以直接访问私有类型的构造方法:private构造/成员方法不能直接调用newInstance创建对象，需要暴力反射
                     //需要遍历
                                            Constructor[] getConstructors();
                                            Constructor[] getDeclareConstructors();

                     反射获取成员方法(Method类):
                                                     Method getMethod("方法名",方法的参数类型.class);
                                                     Method getDeclaredMethod("方法名",方法的参数类型.class);
                                                     //需要遍历
                                                     Method[] getMethods();//获得所有public方法，包括父类
                                                     Method[] getDeclaredMethod();//获得所有private方法，不包括父类
                                                     Object invoke(Object obj, Object... args)
                         例：（有一个Student类 study()静态方法，eat()私有方法，sleep(int hours)公有方法）
                                                    Class c=Student.class;  Class c=Class.forName("类的全限定名");  
                                                    //快速创建对象
                                                    Student stu=(Student)c.newInstance();Class c=stu.getClass();
                                                    //获取公有方法sleep
                                                    Method m1=c.getMethod("sleep",int.class);
                                                     //调用方法
                                                    m1.invoke(stu,8);
                                                    //获取私有方法
                                                    Method m2=c.getDeclaredMethod("eat");
                                                    //私有方法不能直接访问，需要暴力反射
                                                    m2.setAccessible(true);
                                                    //调用私有方法
                                                    m2.invoke(stu);
                                                   //获取静态方法
                                                    Method m3=c.getDeclaredMethod("study");
                                                    //调用静态方法
                                                    m3.invoke(null);
