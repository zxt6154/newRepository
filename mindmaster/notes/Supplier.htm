Supplier(生产者)
          T get() 用来获取一个泛型参数指定类型的对象数据
                用法举例：
                     public static <T>T getObject(Supplier<T> sup){
                                       return sup.get();
                                             }
                      public static void main(String[] args){

//a) 长度为5的String数组

getObj(()->new String[5]);

//b) 包含5个1-20(含1和20)之间随机数的HashSet<Integer>集合

getObj(()->{

HashSet<Integer> set=new HashSet<>();

Random r=new Random();
for (int i = 0; i <5 ; i++) {
Integer num=(r.nextInt(20)+1);
set.add(num);
}
return set;

});

//c) 一个代表2018年4月1日的Calendar对象

getObj(()->{

Calendar c=Calendar.getInstance();//Calendar是抽象类，不能直接使用，常用子类GregorianCalendar，建议调用其静态方法返回子类对象

c.set(Calendar.YEAR,2018);

c.set(Calendar.MONTH,4);

c.set(Calendar.DAY_MONTH,1);

return c;

});
